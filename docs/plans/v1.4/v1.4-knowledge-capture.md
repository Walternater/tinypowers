### 1.4 Knowledge Capture 自动化详细设计

#### extract-knowledge.sh 完整实现

```bash
#!/bin/bash

# 知识提取规则引擎

KNOWLEDGE_FILE="docs/knowledge.md"

# 提取锁定决策
extract_decisions() {
    local spec_file=$1
    
    grep -A 2 "^| D-" "$spec_file" 2>/dev/null | while read -r line; do
        if [[ $line =~ ^\|\ (D-[0-9]+) ]]; then
            local id="${BASH_REMATCH[1]}"
            local decision=$(echo "$line" | cut -d'|' -f3 | tr -d ' ')
            local reason=$(echo "$line" | cut -d'|' -f4)
            
            echo "### 约定"
            echo "**主题**: $decision"
            echo "**约束**: $reason"
            echo "**来源**: $id"
            echo ""
        fi
    done
}

# 提取踩坑记录
extract_pitfalls() {
    local verification_file=$1
    
    # 从验证文档提取问题
    grep -B 2 -A 2 "问题\|踩坑\|坑" "$verification_file" 2>/dev/null | while read -r line; do
        echo "### 踩坑"
        echo "**问题**: $line"
        echo "**记录时间**: $(date +%Y-%m-%d)"
        echo ""
    done
}

# 主逻辑
main() {
    local feature_dir=$1
    local spec_file="$feature_dir/spec.md"
    local verification_file="$feature_dir/VERIFICATION.md"
    
    # 确保知识文件存在
    [ ! -f "$KNOWLEDGE_FILE" ] && echo "# 项目知识沉淀" > "$KNOWLEDGE_FILE"
    
    # 提取并追加
    {
        echo "## $(date +%Y-%m-%d)"
        echo ""
        [ -f "$spec_file" ] && extract_decisions "$spec_file"
        [ -f "$verification_file" ] && extract_pitfalls "$verification_file"
    } >> "$KNOWLEDGE_FILE"
    
    echo "知识已沉淀到 $KNOWLEDGE_FILE"
}

main "$@"
```
