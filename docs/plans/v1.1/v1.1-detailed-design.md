#### check-coverage.sh 完整实现

```bash
#!/bin/bash

RED='\033[0;31m'
GREEN='\033[0;32m'
YELLOW='\033[1;33m'
NC='\033[0m'

# 默认阈值
COVERAGE_THRESHOLD=${COVERAGE_THRESHOLD:-80}

# Maven 覆盖率检查
check_maven_coverage() {
    echo "执行测试并检查覆盖率..."
    
    # 执行测试
    if ! mvn test -q 2>&1; then
        echo -e "${RED}✗ 测试失败${NC}"
        exit 1
    fi
    
    # 检查 JaCoCo 报告
    local report_file="target/site/jacoco/index.html"
    if [ ! -f "$report_file" ]; then
        echo -e "${YELLOW}⚠ 未找到覆盖率报告，跳过检查${NC}"
        return 0
    fi
    
    # 解析覆盖率（简化版，实际需要解析 HTML/XML）
    local coverage=$(grep -oP 'Total[^%]+%' "$report_file" | head -1 | grep -oP '\d+%' || echo "0%")
    local coverage_num=${coverage%\%}
    
    if [ "$coverage_num" -lt "$COVERAGE_THRESHOLD" ]; then
        echo -e "${RED}✗ 覆盖率 $coverage，低于阈值 $COVERAGE_THRESHOLD%${NC}"
        exit 1
    fi
    
    echo -e "${GREEN}✓ 覆盖率 $coverage，满足阈值${NC}"
}

# Gradle 覆盖率检查
check_gradle_coverage() {
    local gradle_cmd="./gradlew"
    [ ! -x "$gradle_cmd" ] && gradle_cmd="gradle"
    
    if ! $gradle_cmd test --quiet 2>&1; then
        echo -e "${RED}✗ 测试失败${NC}"
        exit 1
    fi
    
    # 类似解析逻辑...
    echo -e "${GREEN}✓ 测试通过${NC}"
}

main() {
    if [ -f "pom.xml" ]; then
        check_maven_coverage
    elif [ -f "build.gradle" ]; then
        check_gradle_coverage
    fi
}

main "$@"
```
