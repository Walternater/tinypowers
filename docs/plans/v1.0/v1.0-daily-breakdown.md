### 1.1 工程化门禁详细设计

#### check-compile.sh 完整实现

```bash
#!/bin/bash
set -e

# 颜色定义
RED='\033[0;31m'
GREEN='\033[0;32m'
YELLOW='\033[1;33m'
NC='\033[0m' # No Color

# 检测构建工具
detect_build_tool() {
    if [ -f "pom.xml" ]; then
        echo "maven"
    elif [ -f "build.gradle" ] || [ -f "build.gradle.kts" ]; then
        echo "gradle"
    else
        echo "unknown"
    fi
}

# Maven 编译
check_maven() {
    echo "检测到 Maven 项目，执行编译检查..."
    if ! mvn compile -q -DskipTests 2>&1; then
        echo -e "${RED}✗ 编译失败${NC}"
        echo "请检查："
        echo "  1. mvn -X compile 查看详细错误"
        echo "  2. 依赖是否正确下载"
        exit 1
    fi
    echo -e "${GREEN}✓ Maven 编译通过${NC}"
}

# Gradle 编译
check_gradle() {
    echo "检测到 Gradle 项目，执行编译检查..."
    local gradle_cmd="./gradlew"
    if [ ! -x "$gradle_cmd" ]; then
        gradle_cmd="gradle"
    fi
    
    if ! $gradle_cmd compileJava --quiet 2>&1; then
        echo -e "${RED}✗ 编译失败${NC}"
        echo "请检查："
        echo "  1. ./gradlew compileJava 查看详细错误"
        exit 1
    fi
    echo -e "${GREEN}✓ Gradle 编译通过${NC}"
}

# 主逻辑
main() {
    local build_tool=$(detect_build_tool)
    
    case $build_tool in
        maven)
            check_maven
            ;;
        gradle)
            check_gradle
            ;;
        unknown)
            echo -e "${RED}✗ 未检测到支持的构建工具${NC}"
            echo "需要 pom.xml 或 build.gradle"
            exit 1
            ;;
    esac
}

main "$@"
```

#### check-style.sh 完整实现

```bash
#!/bin/bash

RED='\033[0;31m'
GREEN='\033[0;32m'
YELLOW='\033[1;33m'
NC='\033[0m'

# Maven Spotless
check_maven_spotless() {
    if ! mvn spotless:check -q 2>/dev/null; then
        echo -e "${YELLOW}⚠ 格式化检查失败${NC}"
        echo "自动修复命令: mvn spotless:apply"
        echo ""
        echo "检查详情:"
        mvn spotless:check 2>&1 | grep -A 2 "Violations"
        return 1
    fi
    echo -e "${GREEN}✓ 格式化检查通过${NC}"
    return 0
}

# Gradle Spotless
check_gradle_spotless() {
    local gradle_cmd="./gradlew"
    [ ! -x "$gradle_cmd" ] && gradle_cmd="gradle"
    
    if ! $gradle_cmd spotlessCheck --quiet 2>/dev/null; then
        echo -e "${YELLOW}⚠ 格式化检查失败${NC}"
        echo "自动修复命令: $gradle_cmd spotlessApply"
        return 1
    fi
    echo -e "${GREEN}✓ 格式化检查通过${NC}"
    return 0
}

# 主逻辑
main() {
    if [ -f "pom.xml" ]; then
        check_maven_spotless
    elif [ -f "build.gradle" ]; then
        check_gradle_spotless
    else
        echo -e "${YELLOW}⚠ 未检测到构建工具，跳过格式化检查${NC}"
        exit 0
    fi
}

main "$@"
```

#### check-security.sh 完整实现

```bash
#!/bin/bash

RED='\033[0;31m'
GREEN='\033[0;32m'
YELLOW='\033[1;33m'
NC='\033[0m'

# 检查 Maven OWASP
check_maven_owasp() {
    echo "检查 Maven 依赖漏洞..."
    
    # 确保插件存在
    if ! grep -q "dependency-check-maven" pom.xml 2>/dev/null; then
        echo -e "${YELLOW}⚠ 未配置 OWASP 插件，跳过${NC}"
        return 0
    fi
    
    local output=$(mvn org.owasp:dependency-check-maven:check -q 2>&1)
    
    # 解析高危漏洞
    local high_count=$(echo "$output" | grep -c "High" || echo "0")
    
    if [ "$high_count" -gt 0 ]; then
        echo -e "${RED}✗ 发现 $high_count 个高危依赖漏洞${NC}"
        echo "详情: target/dependency-check-report.html"
        exit 1
    fi
    
    echo -e "${GREEN}✓ 无高危依赖漏洞${NC}"
}

# 检查 Gradle OWASP
check_gradle_owasp() {
    echo "检查 Gradle 依赖漏洞..."
    
    local gradle_cmd="./gradlew"
    [ ! -x "$gradle_cmd" ] && gradle_cmd="gradle"
    
    if ! grep -q "dependency-check" build.gradle 2>/dev/null; then
        echo -e "${YELLOW}⚠ 未配置 OWASP 插件，跳过${NC}"
        return 0
    fi
    
    local output=$($gradle_cmd dependencyCheckAnalyze --quiet 2>&1)
    local high_count=$(echo "$output" | grep -c "HIGH" || echo "0")
    
    if [ "$high_count" -gt 0 ]; then
        echo -e "${RED}✗ 发现 $high_count 个高危依赖漏洞${NC}"
        exit 1
    fi
    
    echo -e "${GREEN}✓ 无高危依赖漏洞${NC}"
}

# 主逻辑
main() {
    if [ -f "pom.xml" ]; then
        check_maven_owasp
    elif [ -f "build.gradle" ]; then
        check_gradle_owasp
    fi
}

main "$@"
```
